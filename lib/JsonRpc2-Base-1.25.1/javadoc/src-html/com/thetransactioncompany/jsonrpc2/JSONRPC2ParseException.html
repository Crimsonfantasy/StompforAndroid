001    package com.thetransactioncompany.jsonrpc2;
002    
003    
004    /** 
005     * Thrown to indicate an exception during the parsing of a JSON-RPC 2.0 
006     * message string.
007     *
008     * <p>The JSON-RPC 2.0 specification and user group forum can be found 
009     * <a href="http://groups.google.com/group/json-rpc">here</a>.
010     *
011     * @author Vladimir Dzhuvinov
012     * @version 1.25.1 (2010-05-18)
013     */
014    public class JSONRPC2ParseException extends Exception {
015            
016            
017            /**
018             * Indicates a parse exception caused by a JSON message not conforming
019             * to the JSON-RPC 2.0 protocol.
020             */
021            public static int PROTOCOL = 0;
022            
023            
024            /**
025             * Indicates a parse exception caused by invalid JSON.
026             */
027            public static int JSON = 1;
028            
029            
030            /**
031             * The parse exception cause type. Default is {@link #PROTOCOL}.
032             */
033            private int causeType = PROTOCOL;
034            
035            
036            
037            /** 
038             * The string that could't be parsed.
039             */
040            private String unparsableString = null;
041            
042            
043            /** 
044             * Creates a new parse exception with the specified message. The cause 
045             * type is set to {@link #PROTOCOL}.
046             *
047             * @param message The exception message.
048             */
049            public JSONRPC2ParseException(final String message) {
050            
051                    super(message);
052            }
053            
054            
055            /**
056             * Creates a new parse exception with the specified message and the 
057             * original string that didn't parse. The cause type is set to
058             * {@link #PROTOCOL}.
059             *
060             * @param message          The exception message.
061             * @param unparsableString The unparsable string.
062             */
063            public JSONRPC2ParseException(final String message, final String unparsableString) {
064            
065                    super(message);
066                    this.unparsableString = unparsableString;
067            }
068            
069            
070            /**
071             * Creates a new parse exception with the specified message, cause type 
072             * and the original string that didn't parse.
073             *
074             * @param message          The exception message.
075             * @param causeType        The exception cause type, either 
076             *                         {@link #PROTOCOL} or {@link #JSON}.
077             * @param unparsableString The unparsable string.
078             */
079            public JSONRPC2ParseException(final String message, final int causeType, final String unparsableString) {
080            
081                    super(message);
082                    
083                    if (causeType != PROTOCOL && causeType != JSON)
084                            throw new IllegalArgumentException("Cause type must be either PROTOCOL or JSON");
085                    
086                    this.causeType = causeType;
087                    this.unparsableString = unparsableString;
088            }
089            
090            
091            /**
092             * Gets the parse exception cause type.
093             *
094             * @return The cause type, either {@link #PROTOCOL} or {@link #JSON}.
095             */
096            public int getCauseType() {
097            
098                    return causeType;
099            }
100            
101            
102            /**
103             * Gets original string that caused the parse exception (if specified).
104             *
105             * @return The string that didn't parse, {@code null} if none.
106             */
107            public String getUnparsableString() {
108            
109                    return unparsableString;
110            }
111    }
