009    /**
010     * Parses JSON-RPC 2.0 request, notification and response messages. 
011     *
012     * <p>Parsing of batched requests / notifications is not supported.
013     *
014     * <p>This class is not thread-safe. A parser instance should not be used by 
015     * more than one thread unless properly synchronised. Alternatively, you may 
016     * use the thread-safe {@link JSONRPC2Message#parse} and its sister methods.
017     *
018     * <p>Example:
019     *
020     * <pre>
021     * String jsonString = "{\"method\":\"makePayment\"," +
022     *                      "\"params\":{\"recipient\":\"Penny Adams\",\"amount\":175.05}," +
023     *                      "\"id\":\"0001\","+
024     *                      "\"jsonrpc\":\"2.0\"}";
025     *  
026     *  JSONRPC2Request req = null;
027     *
028     * JSONRPC2Parser parser = new JSONRPC2Parser();
029     *  
030     *  try {
031     *          req = parser.parseJSONRPC2Request(jsonString);
032     * 
033     *  } catch (JSONRPC2ParseException e) {
034     *          // handle exception
035     *  }
036     *
037     * </pre>
038     *
039     * <p id="map">The mapping between JSON and Java entities (as defined by the 
040     * underlying JSON Smart library): 
041     *
042     * <pre>
043     *     true|false  <--->  java.lang.Boolean
044     *     number      <--->  java.lang.Number
045     *     string      <--->  java.lang.String
046     *     array       <--->  java.util.List
047     *     object      <--->  java.util.Map
048     *     null        <--->  null
049     * </pre>
050     *
051     * <p>The JSON-RPC 2.0 specification and user group forum can be found 
052     * <a href="http://groups.google.com/group/json-rpc">here</a>.
053     * 
054     * @author Vladimir Dzhuvinov
055     * @version 1.25.1 (2011-08-06)
056     */
057    public class JSONRPC2Parser {

099                    }

180            /**
181             * Parses a JSON object string. Provides the initial parsing of JSON-RPC
182             * 2.0 messages. The member order of JSON objects will be preserved if
183             * {@link #preserveOrder} is set to {@code true}.
184             *
185             * @param jsonString The JSON string to parse.
186             *
187             * @return The parsed JSON object.
188             *
189             * @throws JSONRPC2ParseException With detailed message if parsing 
190             *                                failed.
191             */
192            @SuppressWarnings("unchecked")
193            private Map<String,Object> parseJSONObject(final String jsonString)
194                    throws JSONRPC2ParseException {
195            
196                    if (jsonString == null)
197                            throw new JSONRPC2ParseException("Null argument", JSONRPC2ParseException.JSON, null);
198                    
199                    if (jsonString.trim().length()==0)
200                            throw new JSONRPC2ParseException("Invalid JSON: Empty string", JSONRPC2ParseException.JSON, jsonString);
201                    
202                    Object json;
203                    
204                    // Parse the JSON string
205                    try {
206                            if (preserveOrder)
207                                    json = parser.parse(jsonString, linkedContainerFactory);
208                            else
209                                    json = parser.parse(jsonString);
210                                    
211                    } catch (ParseException e) {
212                            // Terse message, do not include full parse exception message
213                            throw new JSONRPC2ParseException("Invalid JSON", JSONRPC2ParseException.JSON, jsonString);
214                    }
215                    
216                    if (json instanceof List)
217                            throw new JSONRPC2ParseException("JSON-RPC 2.0 batch requests/notifications not supported", jsonString);
218                            
219                    if (! (json instanceof Map))
220                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 message: Message must be a JSON object", jsonString);
221                    
222                    return (Map<String,Object>)json;
223            }

226            /**
227             * Ensures the specified parameter is a {@code String} object set to
228             * "2.0". This method is intended to check the "jsonrpc" attribute 
229             * during parsing of JSON-RPC messages.
230             *
231             * @param version    The version parameter.
232             * @param jsonString The original JSON string.
233             *
234             * @throws JSONRPC2Exception If the parameter is not a string matching
235             *                           "2.0".
236             */
237            private static void ensureVersion2(final Object version, final String jsonString)
238                    throws JSONRPC2ParseException {
239            
240                    if (version == null)
241                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0: Version string missing", jsonString);
242                            
243                    else if (! (version instanceof String))
244                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0: Version not a JSON string", jsonString);
245                            
246                    else if (! version.equals("2.0"))
247                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0: Version must be \"2.0\"", jsonString);
248            }

251            /** 
252             * Provides common parsing of JSON-RPC 2.0 requests, notifications 
253             * and responses. Use this method if you don't know which type of 
254             * JSON-RPC message the input string represents.
255             *
256             * <p>If you are certain about the message type use the dedicated 
257             * {@link #parseJSONRPC2Request}, {@link #parseJSONRPC2Notification} 
258             * and {@link #parseJSONRPC2Response} methods. They are more efficient 
259             * and would provide you with more detailed parse error reporting.
260             *
261             * @param jsonString A JSON string representing a JSON-RPC 2.0 request, 
262             *                   notification or response, UTF-8 encoded.
263             *
264             * @return An instance of {@link JSONRPC2Request}, 
265             *         {@link JSONRPC2Notification} or {@link JSONRPC2Response}.
266             *
267             * @throws JSONRPC2ParseException With detailed message if the parsing 
268             *                                failed.
269             */
270            public JSONRPC2Message parseJSONRPC2Message(final String jsonString)
271                    throws JSONRPC2ParseException {
272            
273                    // Try each of the parsers until one succeeds (or all fail)
274                    try {
275                            return parseJSONRPC2Request(jsonString);
276    
277                    } catch (JSONRPC2ParseException e) {
278                    
279                            // throw on JSON error, ignore on protocol error
280                            if (e.getCauseType() == JSONRPC2ParseException.JSON)
281                                    throw e;
282                    }
283                    
284                    try {
285                            return parseJSONRPC2Notification(jsonString);
286                            
287                    } catch (JSONRPC2ParseException e) {
288                            
289                            // throw on JSON error, ignore on protocol error
290                            if (e.getCauseType() == JSONRPC2ParseException.JSON)
291                                    throw e;
292                    }
293                    
294                    try {
295                            return parseJSONRPC2Response(jsonString);
296                            
297                    } catch (JSONRPC2ParseException e) {
298                            
299                            // throw on JSON error, ignore on protocol error
300                            if (e.getCauseType() == JSONRPC2ParseException.JSON)
301                                    throw e;
302                    }
303                    
304                    throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 message", JSONRPC2ParseException.PROTOCOL, jsonString);
305            }

308            /** 
309             * Parses a JSON-RPC 2.0 request string.
310             *
311             * @param jsonString The JSON-RPC 2.0 request string, UTF-8 encoded.
312             *
313             * @return The corresponding JSON-RPC 2.0 request object.
314             *
315             * @throws JSONRPC2ParseException With detailed message if parsing 
316             *                                failed.
317             */
318            public JSONRPC2Request parseJSONRPC2Request(final String jsonString)
319                    throws JSONRPC2ParseException {
320            
321                    // Initial JSON object parsing
322                    Map<String,Object> json = parseJSONObject(jsonString);
323                    
324                    
325                    // Check for JSON-RPC version "2.0"
326                    Object version = json.remove("jsonrpc");
327                    
328                    if (! ignoreVersion)
329                            ensureVersion2(version, jsonString);
330                            
331                    
332                    // Extract method name
333                    Object method = json.remove("method");
334                    
335                    if (method == null)
336                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 request: Method name missing", jsonString);
337                    else if (! (method instanceof String))
338                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 request: Method name not a JSON string", jsonString);
339                    else if (((String)method).length() == 0)
340                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 request: Method name is an empty string", jsonString);
341                    
342                    
343                    // Extract ID
344                    if (! json.containsKey("id"))
345                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 request: Missing identifier", jsonString);
346                    
347                    Object id = json.remove("id");
348                    
349                    if (  id != null             &&
350                        !(id instanceof Number ) &&
351                        !(id instanceof Boolean) &&
352                        !(id instanceof String )    )
353                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 request: Identifier not a JSON scalar", jsonString);
354                    
355                    
356                    // Extract params
357                    Object params = json.remove("params");
358                    
359                    
360                    JSONRPC2Request request = null;
361                    
362                    if (params == null)
363                            request = new JSONRPC2Request((String)method, id);
364                    else if (params instanceof List)
365                            request = new JSONRPC2Request((String)method, (List)params, id);
366                    else if (params instanceof Map)
367                            request = new JSONRPC2Request((String)method, (Map)params, id);
368                    else
369                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 request: Method parameters have unexpected JSON type", jsonString);
370                    
371                    
372                    // Extract remaining non-std params?
373                    if (parseNonStdAttributes) {
374                    
375                            Iterator <Map.Entry<String,Object>> it = json.entrySet().iterator();
376                            
377                            while (it.hasNext()) {
378                                    
379                                    Map.Entry <String,Object> entry = it.next();
380                                    
381                                    request.appendNonStdAttribute(entry.getKey(), entry.getValue());
382                            }
383                    }
384                    
385                    return request;
386            }

389            /** 
390             * Parses a JSON-RPC 2.0 notification string.
391             *
392             * @param jsonString The JSON-RPC 2.0 notification string, UTF-8 
393             *                   encoded.
394             *
395             * @return The corresponding JSON-RPC 2.0 notification object.
396             *
397             * @throws JSONRPC2ParseException With detailed message if parsing 
398             *                                failed.
399             */
400            public JSONRPC2Notification parseJSONRPC2Notification(final String jsonString)
401                    throws JSONRPC2ParseException {
402            
403                    // Initial JSON object parsing
404                    Map <String,Object> json = parseJSONObject(jsonString);
405                    
406                    
407                    // Check for JSON-RPC version "2.0"
408                    Object version = json.remove("jsonrpc");
409                    
410                    if (! ignoreVersion)
411                            ensureVersion2(version, jsonString);
412                    
413                    
414                    // Extract method name
415                    Object method = json.remove("method");
416                    
417                    if (method == null)
418                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 notification: Method name missing", jsonString);
419                    else if (! (method instanceof String))
420                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 notification: Method name not a JSON string", jsonString);
421                    else if (((String)method).length() == 0)
422                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 notification: Method name is an empty string", jsonString);
423                    
424                                    
425                    // Extract params
426                    Object params = json.get("params");
427                    
428                    JSONRPC2Notification notification = null;
429                    
430                    if (params == null)
431                            notification = new JSONRPC2Notification((String)method);
432                    else if (params instanceof List)
433                            notification = new JSONRPC2Notification((String)method, (List)params);
434                    else if (params instanceof Map)
435                            notification = new JSONRPC2Notification((String)method, (Map)params);
436                    else
437                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 notification: Method parameters have unexpected JSON type", jsonString);
438            
439                    // Extract remaining non-std params?
440                    if (parseNonStdAttributes) {
441                    
442                            Iterator <Map.Entry<String,Object>> it = json.entrySet().iterator();
443                            
444                            while (it.hasNext()) {
445                                    
446                                    Map.Entry <String,Object> entry = it.next();
447                                    
448                                    notification.appendNonStdAttribute(entry.getKey(), entry.getValue());
449                            }
450                    }
451                    
452                    return notification;
453            }

456            /** 
457             * Parses a JSON-RPC 2.0 response string.
458             *
459             * @param jsonString The JSON-RPC 2.0 response string, UTF-8 encoded.
460             *
461             * @return The corresponding JSON-RPC 2.0 response object.
462             *
463             * @throws JSONRPC2ParseException With detailed message if parsing 
464             *                                failed.
465             */
466            @SuppressWarnings("unchecked")
467            public JSONRPC2Response parseJSONRPC2Response(final String jsonString)
468                    throws JSONRPC2ParseException {
469            
470                    // Initial JSON object parsing
471                    Map <String,Object> json = parseJSONObject(jsonString);
472                    
473                    // Check for JSON-RPC version "2.0"
474                    Object version = json.remove("jsonrpc");
475                    
476                    if (! ignoreVersion)
477                            ensureVersion2(version, jsonString);
478                    
479                    
480                    // Extract request ID
481                    Object id = json.remove("id");
482                    
483                    if (   id != null             &&
484                        ! (id instanceof Boolean) &&
485                        ! (id instanceof Number ) &&
486                        ! (id instanceof String )    )
487                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 response: Identifier not a JSON scalar", jsonString);
488                    
489                    
490                    // Extract result/error and create response object
491                    // Note: result and error are mutually exclusive
492                    
493                    JSONRPC2Response response = null;
494                    
495                    if (json.containsKey("result") && ! json.containsKey("error")) {
496                            
497                            // Success
498                            Object res = json.remove("result");
499                            
500                            response = new JSONRPC2Response(res, id);
501                                            
502                    }
503                    else if (! json.containsKey("result") && json.containsKey("error")) {
504                    
505                            // Error JSON object
506                            Map <String,Object> errorJSON = (Map<String,Object>)json.remove("error");
507                            
508                            if (errorJSON == null)
509                                    throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 response: Missing error object", jsonString);
510                            
511                            int errorCode;
512                            try {
513                                    errorCode = ((Long)errorJSON.get("code")).intValue();
514                            } catch (Exception e) {
515                                    throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 response: Error code missing or not an integer", jsonString);
516                            }
517                            
518                            String errorMessage = null;
519                            try {
520                                    errorMessage = (String)errorJSON.get("message");
521                            } catch (Exception e) {
522                                    throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 response: Error message missing or not a string", jsonString);
523                            }
524                            
525                            Object errorData = errorJSON.get("data");
526                            
527                            response = new JSONRPC2Response(new JSONRPC2Error(errorCode, errorMessage, errorData), id);
528                            
529                    }
530                    else if (json.containsKey("result") && json.containsKey("error")) {
531                            // Invalid response
532                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 response: You cannot have result and error at the same time", jsonString);
533                    }
534                    else if (! json.containsKey("result") && ! json.containsKey("error")){
535                            // Invalid response
536                            throw new JSONRPC2ParseException("Invalid JSON-RPC 2.0 response: Neither result nor error specified", jsonString);
537                    }
538                    else {
539                            throw new AssertionError();
540                    }
541                    
542                    
543                    // Extract remaining non-std params?
544                    if (parseNonStdAttributes) {
545                    
546                            Iterator <Map.Entry<String,Object>> it = json.entrySet().iterator();
547                            
548                            while (it.hasNext()) {
549                                    
550                                    Map.Entry <String,Object> entry = it.next();
551                                    
552                                    response.appendNonStdAttribute(entry.getKey(), entry.getValue());
553                            }
554                    }
555                    
556                    return response;
557            }
