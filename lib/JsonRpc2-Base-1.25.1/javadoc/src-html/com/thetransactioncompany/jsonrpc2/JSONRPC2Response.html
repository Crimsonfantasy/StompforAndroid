001    package com.thetransactioncompany.jsonrpc2;
002    
003    
004    import java.util.*;
005    
006    import net.minidev.json.JSONObject;
007    
008    
009    /** 
010     * Represents a JSON-RPC 2.0 response.
011     *
012     * <p>A response is returned to the caller after a JSON-RPC 2.0 request has 
013     * been processed (notifications, however, don't produce a response). The
014     * response can take two different forms depending on the outcome:
015     *
016     * <ul>
017     *     <li>The request was successful. The corresponding response returns
018     *         a JSON object with the following information:
019     *         <ul>
020     *             <li>{@code result} The result, which can be of any JSON type
021     *                 - a number, a boolean value, a string, an array, an object 
022     *                 or null.
023     *             <li>{@code id} The request identifier which is echoed back back 
024     *                 to the caller.
025     *             <li>{@code jsonrpc} A string indicating the JSON-RPC protocol 
026     *                 version set to "2.0".
027     *         </ul>
028     *     <li>The request failed. The returned JSON object contains:
029     *         <ul>
030     *             <li>{@code error} An object with:
031     *                 <ul>
032     *                     <li>{@code code} An integer indicating the error type.
033     *                     <li>{@code message} A brief error messsage.
034     *                     <li>{@code data} Optional error data.
035     *                 </ul>
036     *             <li>{@code id} The request identifier. If it couldn't be 
037     *                 determined, e.g. due to a request parse error, the ID is
038     *                 set to {@code null}.
039     *             <li>{@code jsonrpc} A string indicating the JSON-RPC protocol 
040     *                 version set to "2.0".
041     *         </ul>
042     * </ul>
043     *
044     * <p>Here is an example JSON-RPC 2.0 response string where the request
045     * has succeeded:
046     *
047     * <pre>
048     * {  
049     *    "result"  : true,
050     *    "id"      : "req-002",
051     *    "jsonrpc" : "2.0"  
052     * }
053     * </pre>
054     *
055     *
056     * <p>And here is an example JSON-RPC 2.0 response string indicating a failure:
057     *
058     * <pre>
059     * {  
060     *    "error"   : { "code" : -32601, "message" : "Method not found" },
061     *    "id"      : "req-003",
062     *    "jsonrpc" : "2.0"
063     * }
064     * </pre>
065     *
066     * <p>A response object is obtained either by passing a valid JSON-RPC 2.0
067     * response string to the static {@link #parse} method or by invoking the
068     * appropriate constructor.
069     *
070     * <p>Here is how parsing is done:
071     * 
072     * <pre>
073     * String jsonString = "{\"result\":true,\"id\":\"req-002\",\"jsonrpc\":\"2.0\"}";
074     * 
075     * JSONRPC2Response response = null;
076     * 
077     * try {
078     *         response = JSONRPC2Response.parse(jsonString);
079     *
080     * } catch (JSONRPC2Exception e) {
081     *         // handle exception
082     * }
083     * </pre>
084     *
085     * <p>And here is how you can replicate the above example response strings:
086     *
087     * <pre>
088     * // success example
089     * JSONRPC2Response resp = new JSONRPC2Response(true, "req-002");
090     * System.out.println(resp);
091     * 
092     * // failure example
093     * JSONRPC2Error err = new JSONRPC2Error(-32601, "Method not found");
094     * resp = new JSONRPC2Response(err, "req-003");
095     * System.out.println(resp);
096     * 
097     * </pre>
098     *
099     * <p id="map">The mapping between JSON and Java entities (as defined by the 
100     * underlying JSON Smart library): 
101     *
102     * <pre>
103     *     true|false  <--->  java.lang.Boolean
104     *     number      <--->  java.lang.Number
105     *     string      <--->  java.lang.String
106     *     array       <--->  java.util.List
107     *     object      <--->  java.util.Map
108     *     null        <--->  null
109     * </pre>
110     *
111     * <p>The JSON-RPC 2.0 specification and user group forum can be found 
112     * <a href="http://groups.google.com/group/json-rpc">here</a>.
113     *
114     * @author Vladimir Dzhuvinov
115     * @version 1.25.1 (2010-07-10)
116     */
117    public class JSONRPC2Response extends JSONRPC2Message {
118            
119            
120            /** 
121             * The result. 
122             */
123            private Object result = null;
124            
125            
126            /** 
127             * The error object.
128             */
129            private JSONRPC2Error error = null;
130            
131            
132            /** 
133             * The echoed request identifier. 
134             */
135            private Object id = null;
136            
137            
138            /** 
139             * Parses a JSON-RPC 2.0 response string. This method is thread-safe.
140             *
141             * @param jsonString The JSON-RPC 2.0 response string, UTF-8 encoded.
142             *
143             * @return The corresponding JSON-RPC 2.0 response object.
144             *
145             * @throws JSONRPC2ParseException With detailed message if parsing 
146             *                                failed.
147             */
148            public static JSONRPC2Response parse(final String jsonString)
149                    throws JSONRPC2ParseException {
150            
151                    return parse(jsonString, false, false, false);
152            }
153            
154            
155            /** 
156             * Parses a JSON-RPC 2.0 response string. This method is thread-safe.
157             *
158             * @param jsonString    The JSON-RPC 2.0 response string, UTF-8 encoded.
159             * @param preserveOrder {@code true} to preserve the order of JSON 
160             *                      object members in results.
161             *
162             * @return The corresponding JSON-RPC 2.0 response object.
163             *
164             * @throws JSONRPC2ParseException With detailed message if parsing 
165             *                                failed.
166             */
167            public static JSONRPC2Response parse(final String jsonString, final boolean preserveOrder)
168                    throws JSONRPC2ParseException {
169            
170                    return parse(jsonString, preserveOrder, false, false);
171            }
172            
173            
174            /** 
175             * Parses a JSON-RPC 2.0 response string. This method is thread-safe.
176             *
177             * @param jsonString    The JSON-RPC 2.0 response string, UTF-8 encoded.
178             * @param preserveOrder {@code true} to preserve the order of JSON 
179             *                      object members in results.
180             * @param ignoreVersion {@code true} to skip a check of the 
181             *                      {@code "jsonrpc":"2.0"} version attribute in the 
182             *                      JSON-RPC 2.0 message.
183             *
184             * @return The corresponding JSON-RPC 2.0 response object.
185             *
186             * @throws JSONRPC2ParseException With detailed message if the parsing 
187             *                                failed.
188             */
189            public static JSONRPC2Response parse(final String jsonString, final boolean preserveOrder, final boolean ignoreVersion)
190                    throws JSONRPC2ParseException {
191            
192                    return parse(jsonString, preserveOrder, ignoreVersion, false);
193            }
194            
195            
196            /** 
197             * Parses a JSON-RPC 2.0 response string. This method is thread-safe.
198             *
199             * @param jsonString            The JSON-RPC 2.0 response string, UTF-8 
200             *                              encoded.
201             * @param preserveOrder         {@code true} to preserve the order of  
202             *                              JSON object members in results.
203             * @param ignoreVersion         {@code true} to skip a check of the 
204             *                              {@code "jsonrpc":"2.0"} version 
205             *                              attribute in the JSON-RPC 2.0 message.
206             * @param parseNonStdAttributes {@code true} to parse non-standard
207             *                              attributes found in the JSON-RPC 2.0 
208             *                              message.
209             *
210             * @return The corresponding JSON-RPC 2.0 response object.
211             *
212             * @throws JSONRPC2ParseException With detailed message if the parsing 
213             *                                failed.
214             */
215            public static JSONRPC2Response parse(final String jsonString, 
216                                                 final boolean preserveOrder, 
217                                                 final boolean ignoreVersion,
218                                                 final boolean parseNonStdAttributes)
219                    throws JSONRPC2ParseException {
220            
221                    JSONRPC2Parser parser = new JSONRPC2Parser(preserveOrder, ignoreVersion, parseNonStdAttributes);
222                    
223                    return parser.parseJSONRPC2Response(jsonString);
224            }
225            
226            
227            /** 
228             * Creates a new JSON-RPC 2.0 response to a successful request.
229             *
230             * @param result The result. The value can <a href="#map">map</a> 
231             *               to any JSON type.
232             * @param id     The request identifier echoed back to the caller. 
233             */
234            public JSONRPC2Response(final Object result, final Object id) {
235            
236                    setResult(result);
237                    setID(id);
238            }
239            
240            
241            /** 
242             * Creates a new JSON-RPC 2.0 response to a successful request which
243             * result is {@code null}.
244             *
245             * @param id     The request identifier echoed back to the caller. 
246             */
247            public JSONRPC2Response(final Object id) {
248            
249                    setResult(null);
250                    setID(id);
251            }
252            
253            
254            /** 
255             * Creates a new JSON-RPC 2.0 response to a failed request.
256             * 
257             * @param error A JSON-RPC 2.0 error instance indicating the
258             *              cause of the failure.
259             * @param id    The request identifier echoed back to the caller.
260             *              Pass a {@code null} if the request identifier couldn't
261             *              be determined (e.g. due to a parse error).
262             */
263            public JSONRPC2Response(final JSONRPC2Error error, final Object id) {
264            
265                    setError(error);
266                    setID(id);
267            }
268            
269            
270            /** 
271             * Indicates a successful JSON-RPC 2.0 request and sets the result. 
272             * Note that if the response was previously indicating failure this
273             * will turn it into a response indicating success. Any previously set
274             * error data will be invalidated.
275             *
276             * @param result The result. The value can <a href="#map">map</a> to 
277             *               any JSON type.
278             */
279            public void setResult(final Object result) {
280                    
281                    if (   result != null             &&
282                        ! (result instanceof Boolean) &&
283                        ! (result instanceof Number ) &&
284                        ! (result instanceof String ) &&
285                        ! (result instanceof List   ) &&
286                        ! (result instanceof Map    )    )
287                            throw new IllegalArgumentException("The result must map to a JSON type");
288                    
289                    // result and error are mutually exclusive
290                    this.result = result;
291                    this.error = null;
292            }       
293            
294            
295            /** 
296             * Gets the result of the request. The returned value has meaning
297             * only if the request was successful. Use the {@link #getError getError}
298             * method to check this.
299             *
300             * @return The result
301             */
302            public Object getResult() {
303                    
304                    return result;
305            }
306            
307            
308            /** 
309             * Indicates a failed JSON-RPC 2.0 request and sets the error details.
310             * Note that if the response was previously indicating success this
311             * will turn it into a response indicating failure. Any previously set 
312             * result data will be invalidated.
313             *
314             * @param error A JSON-RPC 2.0 error instance indicating the
315             *              cause of the failure.
316             */
317            public void setError(final JSONRPC2Error error) {
318                    
319                    if (error == null)
320                            throw new NullPointerException("The error object cannot be null");
321                    
322                    // result and error are mutually exclusive
323                    this.error = error;
324                    this.result = null;             
325            }
326            
327            
328            /** 
329             * Gets the error object indicating the cause of the request failure. 
330             * If a {@code null} is returned, the request succeeded and there was
331             * no error.
332             *
333             * @return A JSON-RPC 2.0 error object, {@code null} if the
334             *         response indicates success.
335             */
336            public JSONRPC2Error getError() {
337                    
338                    return error;
339            }
340            
341            
342            /**
343             * A convinience method to check if the response indicates success or
344             * failure of the request. Alternatively, you can use the 
345             * {@code #getError} method for this purpose.
346             *
347             * @return {@code true} if the request succeeded, {@code false} if
348             *         there was an error.
349             */
350            public boolean indicatesSuccess() {
351                    
352                    if (error == null)
353                            return true;
354                    else
355                            return false;
356            }
357            
358            
359            /**
360             * Sets the request identifier echoed back to the caller.
361             *
362             * @param id The value must <a href="#map">map</a> to a JSON scalar. 
363             *           Pass a {@code null} if the request identifier couldn't 
364             *           be determined (e.g. due to a parse error).
365             */
366            public void setID(final Object id) {
367                    
368                    if (   id != null             &&
369                        ! (id instanceof Boolean) &&
370                        ! (id instanceof Number ) &&
371                        ! (id instanceof String )    )
372                            throw new IllegalArgumentException("The request identifier must map to a JSON scalar");
373                    
374                    this.id = id;
375            }
376            
377            
378            /** 
379             * Gets the request identifier that is echoed back to the caller.
380             *
381             * @return The request identifier. If there was an error during the
382             *         the request retrieval (e.g. parse error) and the identifier 
383             *         couldn't be determined, the value will be {@code null}.
384             */
385             public Object getID() {
386             
387                    return id;
388            }
389            
390            
391            /** 
392             * Gets a JSON representation of this JSON-RPC 2.0 response.
393             *
394             * @return A JSON object representing the response.
395             */
396            public JSONObject toJSON() {
397                    
398                    JSONObject out = new JSONObject();
399                    
400                    // Result and error are mutually exclusive
401                    if (error != null) {
402                            out.put("error", error.toJSON());
403                    }
404                    else {
405                            out.put("result", result);
406                    }
407                    
408                    out.put("id", id);
409                    
410                    out.put("jsonrpc", "2.0");
411                    
412                    
413                    Map <String,Object> nonStdAttributes = getNonStandardAttributes();
414                    
415                    if (nonStdAttributes != null) {
416                    
417                            Iterator<Map.Entry<String,Object>> it = nonStdAttributes.entrySet().iterator();
418                            
419                            while (it.hasNext()) {
420                            
421                                    Map.Entry <String,Object> pair = it.next();
422                                    out.put(pair.getKey(), pair.getValue());
423                            }
424                    }
425                    
426                    return out;
427            }
428    }
